##### 日志输出

```java

private static final Logger logger = LoggerFactory.getLogger(zhijiandanupJXXMHandler.class);

1. 五个基础方法（最常用）  
- logger.trace(String msg)  
    最轻量、最细的调试级别，一般用于方法级跟踪。生产环境基本不启用。
    
- logger.debug(String msg)  
    调试信息，用于定位问题。开发环境常用，生产一般关闭。
- logger.info(String msg)  
    正常业务流程日志。生产一般开启。
- logger.warn(String msg)  
    警告信息，不影响业务但需要关注。
- logger.error(String msg)  
    严重错误，需要关注或报警。logger.error("xxx", e) 会自动打印堆栈。
1. 带参数的占位写法（最常用）  
    所有方法都支持占位符 {}：  
    logger.info("开始处理项目 ID={}", projectId)  
    logger.warn("数据异常 uid={}, value={}", uid, value)  
    logger.error("保存失败 uid={}", uid, e)  
    优势：延迟 String 拼接，提高性能。
    
2. 带异常对象的方法（非常重要）  
    如果想打印完整异常堆栈，必须用下面这种：  
    logger.error("系统异常", e)  
    注意：logger.error("系统异常 {}", e) 不会打印堆栈，这是误区。
    
3. 其他不太常用但存在的方法
- logger.isDebugEnabled()  
    判断是否开启 debug，提高性能：  
    if (logger.isDebugEnabled()) { logger.debug("调试数据: {}", data) }
    
- logger.atLevel(...)  
    属于新 API，用得较少。
```

##### 字符串转成map对象

```java

/**
     * @param source        原始字符串
     * @param pairSeparator 键值对之间的分隔符 (如 ", " 或 ";")
     * @param kvSeparator   键与值之间的关联符 (如 "=" 或 ":")
     * @return Map对象
     */
    public Map<String, String> stringToMap(String source, String pairSeparator, String kvSeparator) {
        Map<String, String> map = new HashMap<>();

        if (source == null || source.isEmpty()) {
            return map;
        }

        // 1. 去除两端的修饰符（如 { } [ ] 等常见打印格式）
        String cleanStr = source.trim();
        if (cleanStr.startsWith("{") && cleanStr.endsWith("}")) {
            cleanStr = cleanStr.substring(1, cleanStr.length() - 1);
        } else if (cleanStr.startsWith("[") && cleanStr.endsWith("]")) {
            cleanStr = cleanStr.substring(1, cleanStr.length() - 1);
        }

        // 2. 按照键值对分隔符拆分
        // 注意：split支持正则，如果分隔符是特殊字符需注意转义
        String[] pairs = cleanStr.split(pairSeparator);

        for (String pair : pairs) {
            if (pair == null || pair.isEmpty()) continue;

            // 3. 查找第一个关联符的位置（比直接使用 split("=") 性能更好且能处理值中包含等号的情况）
            int separatorIndex = pair.indexOf(kvSeparator);

            if (separatorIndex > 0) {
                String key = pair.substring(0, separatorIndex).trim();
                String val = pair.substring(separatorIndex + 1).trim();

                // 4. 存入Map，忽略重复Key或进行覆盖
                map.put(key, val);
            }
        }

        return map;
    }
```

```java
import com.authine.cloudpivot.engine.open.domain.workflow.WorkflowInstance;

/**
 * 不要直接依赖 WorkflowInstance 上的当前节点字段。
 * 需要先拿流程实例 id，再从 WorkItem 历史里判断节点是否已经流转到目标节点。

private boolean hasAgreedActivity(OpenEngine openEngine, BizObject bizObject, String activityCode) {
    WorkflowInstance workflowInstance = openEngine.getWorkflowInstanceByBizObjectId(bizObject.getId());
    if (workflowInstance == null) {
        return false;
    }

    List<WorkItem> workItems = openEngine.listWorkItems(workflowInstance.getId(), true);

        if (activityCode.equals(workItem.getActivityCode())

            return true;
        }
    }
    return false;
}
```

```java
 /**
     *
     * @param enterDateFormat 输入的日期格式
     * @param outputDateFormat 输出的日期格式

     * @return  返回输出的日期格式字符串

    public String convertToDate(String enterDateFormat, String outputDateFormat, Object dateObj) {
        if (dateObj == null) {
            return null; // 如果日期为空，返回 null
        }
        // 定义日期格式化器
        SimpleDateFormat inputFormatter = new SimpleDateFormat(enterDateFormat); // 输入格式

        try {
            Date date = inputFormatter.parse(dateObj.toString());
            return outputFormatter.format(date);
        } catch (ParseException e) {
            // 捕获异常并返回提示信息
            System.err.println("日期解析失败：" + e.getMessage());
            return null;
        }
    }
```

```java
        if (textDate == null) return null;

                SimpleDateFormat dateFormat = new SimpleDateFormat(format);
```

 /**  
 * 获取指定键名对应的 Map 数据，并提取其 "items" 列表。  
 *  
 * @param map 原始 Map 对象  
 * @param key 键名  
 * @return 提取到的 List<Map<String, Object>> 数据，若不存在则返回 null  
 */public List<Map<String, Object>> getItemsFromMap(Map<String, Object> map, String key) {  
    if (map == null || key == null || key.isEmpty()) {  
        // 参数检查  
        return null;  
    }  
  
    // 获取对应的 Map    Object nestedMap = map.get(key);  
    if (!(nestedMap instanceof Map)) {  
        // 非 Map 类型  
        return null;  
    }  
  
    Map<String, Object> nested = (Map<String, Object>) nestedMap;  
  
    // 获取 "items" 列表  
    Object items = nested.get("items");  
    if (items instanceof List) {  
        // 返回 items 列表  
        return (List<Map<String, Object>>) items;  
    }  
  
    // items 不存在时，尝试返回 id    Object id = nested.get("id");  
    if (id != null) {  
        Map<String, Object> idMap = new HashMap<>();  
        idMap.put("id", id);  
        return Collections.singletonList(idMap);  
    }

##### Object 转换为 BigDecimal

```java

public BigDecimal toBigDecimalWithDynamicDecimal(Object value) {
        if (value == null) {
            return BigDecimal.ZERO;
        }

        try {
            BigDecimal bigDecimal = null;

            // 判断输入类型并转换为 BigDecimal
            if (value instanceof BigDecimal) {
                bigDecimal = (BigDecimal) value;
            } else if (value instanceof String) {
                bigDecimal = new BigDecimal((String) value);
            } else if (value instanceof Double) {
                bigDecimal = BigDecimal.valueOf((Double) value);
            } else if (value instanceof Integer) {
                bigDecimal = BigDecimal.valueOf((Integer) value);
            } else {
                // 如果类型不支持，返回默认值
                return BigDecimal.ZERO;
            }

            // 如果是科学记数法（如 0e8），转成 0
            if (bigDecimal.compareTo(BigDecimal.ZERO) == 0) {
                return BigDecimal.ZERO;
            }

            // 如果是整数，返回整数
            if (bigDecimal.scale() <= 0) {
                return bigDecimal.setScale(0, RoundingMode.HALF_UP);
            }

            // 如果有小数，保持原始小数位数
            int scale = bigDecimal.stripTrailingZeros().scale();
            return bigDecimal.setScale(scale, RoundingMode.HALF_UP);

        } catch (Exception e) {
            // 捕获转换异常，返回默认值
            return BigDecimal.ZERO;
        }
    }
```

```java
        Map<String, String> queryParams = getQueryParams(url);
        return queryParams.getOrDefault(fieldName, null);
    }

    /**
     * 将 URL 中的查询参数解析为键值对
     *
     * @param url 完整的 URL
     * @return 包含查询参数的 Map
     */
    public Map<String, String> getQueryParams(String url) {
        Map<String, String> queryParams = new HashMap<>();
        try {
            URI uri = new URI(url);
            String query = uri.getQuery(); // 获取 URL 查询部分
            if (query != null && !query.isEmpty()) {
                String[] pairs = query.split("&");
                for (String pair : pairs) {
                    String[] keyValue = pair.split("=");
                    String key = URLDecoder.decode(keyValue[0], "UTF-8");
                    String value = keyValue.length > 1 ? URLDecoder.decode(keyValue[1], "UTF-8") : "";
                    queryParams.put(key, value);
                }
            }
        } catch (Exception e) {
            e.printStackTrace(); // 捕获解析 URL 时可能发生的异常
        }
        return queryParams;
    }
```

##### 将数字转成中文数字

```java

 public final String[] CHINESE_NUMS = {
            "零", "一", "二", "三", "四", "五", "六", "七", "八", "九"
    };
    public final String[] CHINESE_UNITS = {
            "", "十", "百", "千", "万", "亿"
    };

    public String numberToChinese(long num) {
        if (num == 0) {
            return CHINESE_NUMS[0];
        }

        StringBuilder result = new StringBuilder();
        int unitPos = 0;  // 当前单位位置
        boolean zeroFlag = false;  // 标记是否出现零

        while (num > 0) {
            int digit = (int) (num % 10);  // 取个位数字
            if (digit == 0) {
                if (!zeroFlag && unitPos > 0) {
                    result.insert(0, CHINESE_NUMS[digit]);
                }
                zeroFlag = true;  // 下一次出现零时可以继续加零
            } else {
                result.insert(0, CHINESE_NUMS[digit] + CHINESE_UNITS[unitPos]);
                zeroFlag = false;  // 出现非零数字时，零标记清除
            }
            num /= 10;
            unitPos++;
        }

        // 处理“十”前面的“零”不显示的问题
        if (result.toString().startsWith("一十")) {
            result.delete(0, 1);  // 如果以“一十”开头，删除“一”
        }

        return result.toString();
    }
```
