```javascript

  // 新增数据事件
    const addTask = () => {
        // 获取选中的任务
        let targetTask = project.getSelected();
        console.log('选中的数据', targetTask);

        // 判断是否选中了有效任务，判断条件为选中的任务存在且 pid 不为 null
        if (targetTask && targetTask['pid'] != null) {
            // 创建一个新的任务对象
            let newTask = project.newTask();

            // 为新任务赋值，给新任务设置相关字段
            newTask['UID'] = getUUID(); // 获取一个随机的 UUID
            newTask['Name'] = '';
            newTask['pid'] = targetTask['pid']; // 新任务的父任务 pid 与目标任务相同
            newTask['id'] = newTask['UID']; // 新任务的 id 使用新生成的 UID
            newTask['branchSubitemCode'] = "";
            newTask['docDirectoryTreeCode'] = "";
            newTask['userCode'] = ""; // 新任务的用户编码，暂时为空
            newTask['branchSubitemName'] = "";
            newTask['sequenceCoding'] = "";
            newTask['joinProjectEngineeringID'] = targetTask['joinProjectEngineeringID']; // 新任务的项目工程 ID 继承父任务的工程 ID
            newTask['unitPro'] = targetTask['unitPro']; // 新任务的单位工程继承父任务
            newTask['branchPro'] = targetTask['branchPro']; // 新任务的分部工程继承父任务
            newTask['subentryPro'] = "";
            newTask['joinbranchSubitemVersion'] = targetTask['joinbranchSubitemVersion']; // 新任务的版本
            newTask['joinBranchSubitem'] = targetTask['joinBranchSubitem']; // 新任务的分部结构继承父任务
            newTask['parentFBFX'] = targetTask['parentFBFX']; // 新任务的父级 FBFX

            console.log("新增的数据", newTask);

            // 将新任务添加到目标任务的后面
            project.addTask(newTask, 'after', targetTask);

            // 重新获取任务树
            let taskTree = project.getTaskTree();

            // 遍历任务树，更新相关字段
            taskTree.forEach(node => {
                // 如果节点的 id 匹配父任务的 pid 并且该节点有子任务
                if (node.id === targetTask['pid'] && node.children) {
                    // 遍历子任务，找到新添加的任务并构建相关字段
                    node.children.forEach(nodeChildren => {
                        // 如果子任务的 UID 与新任务的 UID 匹配
                        if (nodeChildren.UID === newTask['UID']) {
                            // 调用 constructFields 函数来构建 Name 和 sequenceCoding 字段
                            constructFields(node, nodeChildren);

                            // 如果当前子任务的 OutlineNumber 比新任务的 OutlineNumber 大，则更新该子任务
                            node.children.forEach(nodeChildrens => {
                                if (Number(nodeChildrens.OutlineNumber.split('.').pop()) > Number(nodeChildren.OutlineNumber.split('.').pop())) {
                                    constructFields(node, nodeChildrens);
                                }
                            });
                        }
                    });
                }
            });
            //取消选中所有任务。
            project.deselectAll();
        } else {
            // 如果没有选中任务，弹出警告
            that.$message.warning('请选择子项新增数据');
        }
    };

    /**
     * 构建 Name 和 sequenceCoding 字段
     * @param {Object} parentTask - 父任务对象，包含父任务的信息
     * @param {Object} currentTask - 当前任务对象，需要构建 Name 和 sequenceCoding 的任务
     * @returns {void} - 该函数没有返回值，直接更新当前任务对象
     */
    function constructFields(parentTask, currentTask) {
        // 获取父项的 Name 字段
        const parentName = parentTask.Name;

        // 获取当前项的 OutlineNumber 字段值，并将其最后一部分转换为数字
        const currentId = Number(currentTask.OutlineNumber.split('.').pop());

        // 构建新的 Name 字段：由父任务 Name 和当前任务的 ID（OutlineNumber 最后一部分）组成
        const newName = `${parentName}.${currentId}`;

        // 构建 sequenceCoding 字段：由新的 Name 字段拆分成数组，对每部分进行补零，最后拼接成一个字符串
        const sequenceCoding = newName.split('.').map(part => part.padStart(3, '0')).join('');

        // 更新当前任务对象，将新的 Name、docDirectoryTreeCode 和 sequenceCoding 写入任务字段
        project.updateTask(currentTask, {
            Name: newName,
            docDirectoryTreeCode: newName, // docDirectoryTreeCode 使用与 Name 相同的值
            sequenceCoding: sequenceCoding, // 更新序列编码
            branchSubitemCode: sequenceCoding // 更新分支子项编码
        });
    }
```

```javascript
    function lisentClick(that) {

                });

                for (let i = 0; i < zzexcel.SheetNames.length; i++) {
                    const newData = XLSX.utils.sheet_to_json(
                        zzexcel.Sheets[zzexcel.SheetNames[i]]
                    );
                    result.push(...newData);
                }

                let isError = false;
                try {
                    const menuList = {
                        '序号': 'Name',
                        '系统编码': 'branchSubitemCode',
                        '用户编码': 'userCode',
                        '名称': 'branchSubitemName',
                        '排序编码': 'sequenceCoding'
                    };

                    // 数据处理
                    result.forEach((item, index) => {
                        for (let i in item) {
                            if (Object.keys(menuList).indexOf(i) == -1) {
                                that.$message.warning("需要按照模板字段导入数据");
                                throw new Error("抛出错误");
                            }
                            item[menuList[i]] = item[i];
                            delete item[i];  // 删除 item 中的原始字段
                        }
                    });
                } catch (e) {
                    return;
                }

                // 用来存储父项 UID 的映射
                let parentMap = {};
                // 为每条数据增加 UID，id 和 pid
                result.forEach((item) => {
                    // 生成 UID
                    item["UID"] = getUUID();
                    item["id"] = item["UID"];  // id = UID
                    item["docDirectoryTreeCode"] = item["Name"];
                    item["joinProjectEngineeringID"] = that.joinProjectEngineering.value.id;//工程
                    item["unitPro"] = that.joinFBFX.value.id;//单位/子单位工程

                    // 获取 Name 中的父项部分
                    const nameParts = item.Name.split(".");

                    // 处理父项
                    if (nameParts.length > 1) {
                        // 获取父项的UID，如果存在则设为父项的 UID
                        const parentName = nameParts.slice(0, -1).join(".");
                        const parentUID = parentMap[parentName];

                        if (parentUID) {
                            item["pid"] = parentUID;  // 子项的 pid 为父项的 UID
                        } else {
                            item["pid"] = null;  // 如果没有父项，pid 为空
                        }
                    } else {
                        item["pid"] = null;  // 根项的 pid 为空
                    }

                    // 将当前项的 UID 存入父项映射中
                    parentMap[item.Name] = item["UID"];
                });
                console.log("导入的文件数据", result);

                plusganttLoad(that, result)

            };
        });
    }
```

```javascript
    const timeFormatter = (date) => {
        let now = new Date(date);
        let year = now.getFullYear();
        let month = now.getMonth() + 1;
        let day = now.getDate();
```

```javascript
    function flattenTree(tree) {
        let flatArray = [];

        // 递归遍历树形结构
        function flattenNode(node) {
            // 这里可以对节点进行处理，例：添加UID或其他字段
            flatArray.push({ ...node }); // 浅拷贝节点，避免修改原数据
            if (node.children && node.children.length > 0) {
                node.children.forEach(flattenNode); // 遍历子节点
            }
        }

        tree.forEach(flattenNode);
        return flatArray;
    }
```
