import com.authine.cloudpivot.engine.enums.type.BizPropertyType;
import com.authine.cloudpivot.engine.open.OpenEngineContext;
import com.authine.cloudpivot.engine.open.service.OpenEngine;
import com.authine.cloudpivot.engine.service.bizrule.handler.context.Request;
import org.apache.commons.lang3.StringUtils;

import java.math.RoundingMode;

public class signedProjectContractLisgetSignedProjectContractHandler extends BaseHandler {
    BizObject bizObject = null;
    Map<String, Object> params = null;
    OpenEngine openEngine = null;
    String[] possibleFormats = {
            "yyyy-MM-dd",
            "yyyy-MM-dd HH:mm:ss.SSS",
            "yyyy-MM-dd HH:mm:ss",
            "yyyy/MM/dd HH:mm:ss"
            // 可以根据需要添加更多格式
    };

    @Override
    public Object handle(Request request) {
        bizObject = request.getData().getBizObject();
        params = request.getData().getCustomParams();
        openEngine = this.getOpenEngine(request);
        if (params.containsKey("userId") && StringUtils.isNotEmpty(getString(params, "userId"))) {
            OpenEngineContext openContext = new OpenEngineContext();
            openContext.setCurrentUserId((String) params.get("userId"));
            openEngine.setContext(openContext);
        }
        try {
            //外部按钮点击传值触发
            getSignedProjectContract(getString(params, "proid"));
            return "正常";
        } catch (Exception e) {
            System.out.println("错误：" + e);
            return "失败";
        }

    }

    public void getSignedProjectContract(String proid) {
        List<Object> consolidatedContractData = getInvokeBizServiceMethod("signedContractProj", "getsignedProjectContractData", new HashMap<String, Object>() {{
            put("proid", proid); // 建设工程id
        }});
        if (consolidatedContractData != null && consolidatedContractData.size() > 0) {
            for (Object ccdObj : consolidatedContractData) {
                Map<String, Object> ebdlMap = (Map<String, Object>) ccdObj;
                openEngine.deleteBizObject("signedProjectContractLis", getString(ebdlMap, "id"));
            }
        }
        //存数据的地方
        //，合同，合同名称，单位，合同金额，合同（预计）结算金额（元），合同累计结算金额（元），合同项目累计结算金额（元）
        List<Object> table = getInvokeBizServiceMethod("signedContractProj", "getEngineeringContracData", new HashMap<String, Object>() {{
            put("proid", proid); // 建设工程id
        }});
        if (table != null && table.size() > 0) {
            for (int i = 0; i < table.size(); i++) {
                Map<String, Object> tableMap = (Map<String, Object>) table.get(i);
                Map<String, Object> dataMap = new HashMap<>();
                dataMap.put("joinProjectEngineering", buildPropertyMap("projectEngineerinTree", "建设工程树形", "name", proid));
                dataMap.put("joinProjectEngineeringId", proid);
                dataMap.put("joinEngineeringContract", buildPropertyMap("engineeringContrac", "合同库", "name", getString(tableMap, "id")));
                dataMap.put("joinEngineeringContractId", getString(tableMap, "id"));
                dataMap.put("contractState", getString(tableMap, "contractStatus"));
                dataMap.put("contractNumber", getString(tableMap, "contractNumber"));
                dataMap.put("serviceContractName", getString(tableMap, "contractName"));
                dataMap.put("contractorUnit", getString(tableMap, "contractorUnit"));
                dataMap.put("signingDate", getDateTime(tableMap, "signingDate", possibleFormats));
                dataMap.put("contractAmount", getFullNumberString(tableMap, "contractAmount"));
                dataMap.put("contractEstSettleAmount", getFullNumberString(tableMap, "contractEstSettleAmount"));
                dataMap.put("contractTotalSettleAmount", getFullNumberString(tableMap, "contractTotalSettleAmount"));
                dataMap.put("state", "有效");
                dataMap.put("describe", "通过定时器定时更新,在" + formatCurrentDate("yyyy年MM月dd日 HH时mm分ss秒") + "将数据更新为有效和当前");
                //获取子表概算数据
                List<Map<String, Object>> childList = new ArrayList<>();
                List<Object> childTable = getInvokeBizServiceMethod("signedContractProj", "getBudgetManagementSubject", new HashMap<String, Object>() {{
                    put("proid", proid); // 建设工程id
                    put("id", getString(tableMap, "id")); // 合同id
                }});
                //等待累加的值
                BigDecimal acontractProjectAmount = BigDecimal.ZERO;//合同项目金额（元）
                BigDecimal contractProjEstSettleAmount = BigDecimal.ZERO;//合同项目（预计）结算金额（元）
                BigDecimal contractProjTotalSettleAmt = BigDecimal.ZERO;//合同项目累计结算金额（元）
                if (childTable != null && childTable.size() > 0) {
                    for (int j = 0; j < childTable.size(); j++) {
                        Map<String, Object> childTableMap = (Map<String, Object>) childTable.get(j);
                        Map<String, Object> childDataMap = new HashMap<>();
                        childDataMap.put("joinBudgetaryEstimates", buildPropertyMap("budgetaryEstimates", "实例概算管理科目", "name", getString(childTableMap, "joinBudgetaryEstimates")));
                        childDataMap.put("budgetaryEstimatesId", getString(childTableMap, "joinBudgetaryEstimates"));
                        childDataMap.put("joinProjectEngineeringId", proid);
                        childDataMap.put("estimatesName", getString(childTableMap, "estimatesName"));
                        childDataMap.put("sequenceCoding", getString(childTableMap, "sequenceCoding"));
                        childDataMap.put("acontractProjectAmount", getFullNumberString(childTableMap, "acontractProjectAmount"));
                        childDataMap.put("contractProjEstSettleAmount", getFullNumberString(childTableMap, "contractProjEstSettleAmount"));
                        childDataMap.put("contractProjTotalSettleAmt", getFullNumberString(childTableMap, "contractProjTotalSettleAmt"));
                        childDataMap.put("planBasicPrepaid", getFullNumberString(childTableMap, "planBasicPrepaid"));
                        childDataMap.put("actualBasicPrepaid", getFullNumberString(childTableMap, "actualBasicPrepaid"));
                        childDataMap.put("planSpreadsPrepaid", getFullNumberString(childTableMap, "planSpreadsPrepaid"));
                        childDataMap.put("ActualSpreadsPrepaid", getFullNumberString(childTableMap, "ActualSpreadsPrepaid"));
                        acontractProjectAmount = acontractProjectAmount.add(toBigDecimalWithTwoDecimal(getFullNumberString(childTableMap, "acontractProjectAmount")));
                        contractProjEstSettleAmount = contractProjEstSettleAmount.add(toBigDecimalWithTwoDecimal(getFullNumberString(childTableMap, "contractProjEstSettleAmount")));
                        contractProjTotalSettleAmt = contractProjTotalSettleAmt.add(toBigDecimalWithTwoDecimal(getFullNumberString(childTableMap, "contractProjTotalSettleAmt")));
                        childDataMap.put("rowStatus", "Added");
                        childList.add(childDataMap);
                    }
                }
                //已签订合同的数据
                List<Object> signedProjectContractList = getInvokeBizServiceMethod("signedContractProj", "getDetailedBudgetData", new HashMap<String, Object>() {{
                    put("proid", proid); // 建设工程id
                    put("id", getString(tableMap, "id")); // 合同id
                }});
                List<Map<String, Object>> currentContractAmountList1 = new ArrayList<>();
                if (signedProjectContractList != null && signedProjectContractList.size() > 0) {
                    for (int j = 0; j < signedProjectContractList.size(); j++) {
                        Map<String, Object> signedProjectContractMap = (Map<String, Object>) signedProjectContractList.get(j);
                        Map<String, Object> map = new HashMap<>();
                        map.put("budgetaryEstimatesId", signedProjectContractMap.get("joinBudgetaryEstimates"));
                        map.put("estimatesName", signedProjectContractMap.get("estimatesName"));
                        map.put("joinProjectEngineeringId", signedProjectContractMap.get("joinProjectEngineering"));
                        map.put("serviceContractName", signedProjectContractMap.get("projName"));
                        map.put("joinEngineeringContract", getString(tableMap, "id"));
                        map.put("projWBSNo", signedProjectContractMap.get("projWBSNo"));
                        map.put("contractId", signedProjectContractMap.get("id"));
                        map.put("estimatedItems", toBigDecimalWithTwoDecimal(signedProjectContractMap.get("acontractProjectAmount")).toPlainString());
                        map.put("rowStatus", "Added");
                        currentContractAmountList1.add(map);
                    }
                }

                dataMap.put("acontractProjectAmount", acontractProjectAmount.toString());
                dataMap.put("contractProjEstSettleAmount", contractProjEstSettleAmount.toString());
                dataMap.put("contractProjTotalSettleAmt", contractProjTotalSettleAmt.toString());
                dataMap.put("signedProjectContract", childList);
                dataMap.put("currentContractAmountList1", currentContractAmountList1);
                BizObject bo = new BizObject("signedProjectContractLis", null, dataMap);
                openEngine.createBizObject(bo);

            }
        }

    }

    public List<Object> getInvokeBizServiceMethod(String serviceCode, String methodCode, Map<String, Object> args) {
        Object result = openEngine.invokeBizServiceMethod(
                bizObject.getSchemaCode(),
                bizObject.getId(),
                serviceCode,
                methodCode,
                args,
                1000);
        Map<String, Object> resultMap = (Map<String, Object>) result;
        return (List<Object>) resultMap.get("data");
    }

    public Map<String, Object> buildPropertyMap(String schemaCode, String shortText, String displayCode, Object id) {
        Map<String, Object> propertyMap = new HashMap<>();
        propertyMap.put("schemaCode", schemaCode);
        propertyMap.put("ShortText", shortText);
        propertyMap.put("propertyType", BizPropertyType.WORK_SHEET);
        propertyMap.put("displayCode", displayCode);
        propertyMap.put("id", id);
        return propertyMap;
    }

    public String formatCurrentDate(String format) {
        if (format == null || format.isEmpty()) {
            throw new IllegalArgumentException("日期格式不能为空");
        }
        SimpleDateFormat sdf = new SimpleDateFormat(format);
        Date currentDate = new Date(); // 获取当前日期
        return sdf.format(currentDate);
    }

    public String getString(Map<String, Object> data, String key) {
        Object value = data.get(key);
        if (value == null) {
            return "";
        } else {
            return value.toString();
        }
    }

    public String getFullNumberString(Map<String, Object> map, String key) {
        // 检查 map 是否包含 key
        if (map == null || !map.containsKey(key)) {
            throw new IllegalArgumentException("The provided key does not exist in the map.");
        }
        Object value = map.get(key);
        // 如果值为 null，直接返回空字符串
        if (value == null) {
            return "0";
        }
        try {
            // 转换值为 Double
            Double doubleValue = value instanceof Double ? (Double) value : Double.valueOf(value.toString());

            // 使用 DecimalFormat 格式化为完整数值
            DecimalFormat decimalFormat = new DecimalFormat("0.####################");
            return decimalFormat.format(doubleValue);
        } catch (NumberFormatException e) {
            throw new IllegalArgumentException("The value corresponding to the key is not a valid number.", e);
        }
    }

    public Timestamp getDateTime(Map<String, Object> data, String key, String[] possibleFormats) {
        Object value = data.get(key);
        String textDate = value != null ? value.toString() : null;
        if (textDate == null) return null;
        for (String format : possibleFormats) {
            try {
                SimpleDateFormat dateFormat = new SimpleDateFormat(format);
                Date parsedDate = dateFormat.parse(textDate);
                return new Timestamp(parsedDate.getTime());
            } catch (ParseException ignored) {
            }
        }
        return null;
    }

    public BigDecimal toBigDecimalWithTwoDecimal(Object value) {
        if (value == null) {
            return BigDecimal.ZERO;
        }

        try {
            BigDecimal bigDecimal = null;

            // 判断输入类型并转换为 BigDecimal
            if (value instanceof BigDecimal) {
                bigDecimal = (BigDecimal) value;
            } else if (value instanceof String) {
                bigDecimal = new BigDecimal((String) value);
            } else if (value instanceof Double) {
                bigDecimal = BigDecimal.valueOf((Double) value);
            } else if (value instanceof Integer) {
                bigDecimal = BigDecimal.valueOf((Integer) value);
            } else {
                // 如果类型不支持，返回默认值
                return BigDecimal.ZERO;
            }

            // 如果是科学记数法（如 0e8），转成 0
            if (bigDecimal.compareTo(BigDecimal.ZERO) == 0) {
                return BigDecimal.ZERO.setScale(2, RoundingMode.HALF_UP);
            }

            // 保留两位小数
            return bigDecimal.setScale(2, RoundingMode.HALF_UP);

        } catch (Exception e) {
            // 捕获转换异常，返回默认值
            return BigDecimal.ZERO;
        }
    }
}
