SELECT
  a.joinBudgetaryEstimates,
  b.estimatesName,
  b.sequenceCoding,
  COALESCE( SUM( a.projTZJR ), 0 ) + COALESCE( SUM( d.projTZJR1 ), 0 )-  COALESCE( SUM( a.planBasicPrepaid ), 0 )-COALESCE( SUM( a.planSpreadsPrepaid ), 0 ) AS acontractProjectAmount,
  COALESCE( c.settlementAmount, 0 )+ COALESCE( c.settlementAmount, 0 )- COALESCE( SUM( a.actualBasicPrepaid ), 0 ) -COALESCE( SUM( a.ActualSpreadsPrepaid ), 0 )  AS contractProjTotalSettleAmt,
CASE
    WHEN h.contractState = '已结算' THEN
    COALESCE( c.settlementAmount, 0 ) ELSE COALESCE( SUM( a.projTZJR ), 0 ) + COALESCE( SUM( d.projTZJR1 ), 0 ) 
  END AS contractProjEstSettleAmount ,
    COALESCE( SUM( a.planBasicPrepaid ), 0 )  AS planBasicPrepaid,
    COALESCE( SUM( a.actualBasicPrepaid ), 0 )  AS actualBasicPrepaid,
    COALESCE( SUM( a.planSpreadsPrepaid ), 0 )  AS planSpreadsPrepaid,
    COALESCE( SUM( a.ActualSpreadsPrepaid ), 0 )  AS ActualSpreadsPrepaid

FROM
  icci_contractlist a
  JOIN icci_budgetaryestimates b ON a.joinBudgetaryEstimates = b.id 
  AND b.state = '有效'
  LEFT JOIN icci_engineeringcontrac e ON e.id = :id
  AND e.state = '有效'
  LEFT JOIN icci_contractstate h ON e.joinContractState = h.id 
  AND h.state = '有效'
  LEFT JOIN (
    SELECT
      bd.parentId,
      COALESCE( SUM( bd.projTZJR1 ), 0 ) projTZJR1 
    FROM
      icci_contractlist ad
      JOIN icci_contractlistdetailed bd ON bd.parentId = ad.id
    WHERE
      ad.joinEngineeringContract = :id
      AND ad.joinProjectEngineering = :proid
    GROUP BY
      bd.parentId 
  ) d ON d.parentId = a.id
  LEFT JOIN ( 
  SELECT joinBudgetaryEstimates, SUM( settlementAmount ) AS settlementAmount 
  FROM icci_settlementstandingbook 
  WHERE state = '有效'   AND sequenceStatus in  ('PROCESSING','COMPLETED')
 AND joinProjectEngineering = :proid AND joinEngineeringContract = :id 
 GROUP BY joinBudgetaryEstimates ) c 
  ON c.joinBudgetaryEstimates = a.joinBudgetaryEstimates 
 LEFT JOIN ( 
  SELECT joinBudgetaryEstimates, SUM( paymentSdmoney ) AS paymentSdmoney 
  FROM icci_paymentsliplist 
  WHERE state = '有效'   AND sequenceStatus in  ('PROCESSING','COMPLETED')
 AND joinProjectEngineering = :proid AND joinEngineeringContract = :id 
 GROUP BY joinBudgetaryEstimates ) f 
  ON f.joinBudgetaryEstimates = a.joinBudgetaryEstimates 
WHERE
  a.joinEngineeringContract = :id
  AND a.joinProjectEngineering = :proid
  AND a.state = '有效' 
  AND a.childFlag = TRUE 
  AND a.sequenceStatus in  ('PROCESSING','COMPLETED')
  AND b.isLastLevel = TRUE
  AND b.state = '有效' 
  AND b.estimatesName != '不计科目' 
  AND b.versions = '2024新版概算' 
GROUP BY
  a.joinBudgetaryEstimates 
ORDER BY
  b.sequenceCoding
