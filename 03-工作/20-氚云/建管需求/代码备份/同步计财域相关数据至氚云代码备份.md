    /**
     * 优化后的通用同步逻辑：单次映射，直接转换
     */
    private void doSync(String apiName, String methodCode, String schemaCode, int mappingType) {
        logger.info("开始同步: " + apiName);
        long start = System.currentTimeMillis();
        try {
            List<Object> dataList = getInvokeBizServiceMethod("financialDomainServices", methodCode, new HashMap<>());
            if (dataList == null || dataList.isEmpty()) {
                logger.warn(apiName + " 未获取到数据");
                return;
            }

            Map<String, String> fieldMapping = getMapping(mappingType);
            // 预初始化 List 容量，避免频繁扩容
            List<String> bizObjectJsonList = new ArrayList<>(BATCH_SIZE);

            for (Object item : dataList) {
                Map<String, Object> tableMap = (Map<String, Object>) item;
                if (tableMap == null) continue;

                // --- 核心优化点：一次循环完成映射和转换 ---
                JSONObject target = new JSONObject();
                for (Map.Entry<String, String> entry : fieldMapping.entrySet()) {
                    // entry.getKey() 是目标字段，entry.getValue() 是原始字段
                    Object val = tableMap.get(entry.getValue());
                    if (val != null) {
                        target.put(entry.getKey(), val.toString());
                    }
                }

                if (!target.isEmpty()) {
                    bizObjectJsonList.add(target.toJSONString());
                }

                // 满 1000 条即发送
                if (bizObjectJsonList.size() >= BATCH_SIZE) {
                    flushToCy(bizObjectJsonList, schemaCode);
                    bizObjectJsonList.clear();
                }
            }

            // 处理尾部剩余数据
            if (!bizObjectJsonList.isEmpty()) {
                flushToCy(bizObjectJsonList, schemaCode);
            }

            logger.info(apiName + " 同步成功，总量: " + dataList.size() + "，耗时: " + (System.currentTimeMillis() - start) + "ms");
        } catch (Exception e) {
            logger.error(apiName + " 同步异常", e);
        }
    }

    public void addLog(String code, String type, String data) {
        String nowStr = LocalDateTime.now().format(DatePattern.NORM_DATETIME_FORMATTER);
        Map<String, Object> bizData = new HashMap<>();
        bizData.put("ShortText1765176302991", code);
        bizData.put("ShortText1765176294141", nowStr);
        bizData.put("ShortText1765176314572", type);

        if (data != null && !data.isEmpty()) {
            bizData.put("logContent", data);
            int maxLen = 2000000;
            for (int i = 0; i < data.length(); i += maxLen) {
                int end = Math.min(i + maxLen, data.length());
                bizData.put("msg" + ((i / maxLen) + 1), data.substring(i, end));
            }
        } else {
            bizData.put("msg1", "获取数据为空！！");
        }

        openEngine.createBizObject(new BizObject("FinanceOrderStatsApi", null, bizData));
    }

        }
        return Collections.emptyList();
    }
